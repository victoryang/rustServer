# rust_server

# Get Source
    git clone https://github.com/victoryang/rustServer.git

# Install
    chmod +x pre-install.sh
    . ./pre-install.sh

# Build
    source $HOME/.cargo/env && make

# Get Binary
    rust_server

# Debug
	RUST_BACKTRACE=1
